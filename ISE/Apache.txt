nano /etc/apache2/sites-available/www.uhr.dmz.conf


<VirtualHost *:80>
        ServerName www.uhr.dmz
        ServerAdmin root@localhost

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

</VirtualHost>
